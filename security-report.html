Critical SEC-001 Open CORS — อนุญาตทุก origin โดยไม่มีข้อจำกัด backend/app.js:9 ▼
ประเภท
CORS Misconfiguration
สาเหตุ
app.use(cors()) โดยไม่ระบุ origin หมายความว่า browser ทุกโดเมนสามารถส่ง request มาที่ API ได้ ทำให้ผู้ไม่ประสงค์ดีสร้างเว็บปลอมที่เรียก API ในนามของ user ได้ (CSRF attack)
ผลกระทบ
ข้อมูลงานทั้งหมดสามารถถูกอ่าน แก้ไข หรือลบจากเว็บไซต์ภายนอกได้ โดยผู้ใช้ไม่รู้ตัว

วิธีแก้ไขที่แนะนำ

จำกัด origin เฉพาะที่อนุญาต และอ่านค่าจาก environment variable:

// backend/app.js
const ALLOWED_ORIGINS = (process.env.ALLOWED_ORIGINS || 'http://localhost:5173').split(',')

app.use(cors({
  origin: (origin, callback) => {
    if (!origin || ALLOWED_ORIGINS.includes(origin)) callback(null, true)
    else callback(new Error('Not allowed by CORS'))
  },
  credentials: true
}))

เพิ่ม ALLOWED_ORIGINS=https://yourdomain.com ใน .env

// frontend/src/api/taskApi.js — wrapper ส่วนกลาง
async function apiFetch(url, options = {}) {
  const res = await fetch(url, options)
  if (!res.ok) {
    const body = await res.json().catch(() => ({}))
    throw new Error(body.error || `HTTP ${res.status}`)
  }
  return res.json()
}

// ใน App.jsx
const handleCreate = async (taskData) => {
  try {
    const newTask = await apiFetch('/api/tasks', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(taskData)
    })
    setTasks(prev => [...prev, newTask])
    showToast('เพิ่มงานสำเร็จ', 'success')
  } catch (err) {
    showToast(err.message, 'error')
  }
}
